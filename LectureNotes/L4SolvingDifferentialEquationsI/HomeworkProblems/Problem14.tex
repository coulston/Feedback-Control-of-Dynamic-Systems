Find the solution to the following differential equations by using Laplace Transforms \vspace{2pt}
\begin{enumerate}
\setlength{\itemsep}{8pt}
\item 
\[
\ddot{x} + 2 \dot{x} +1x = 1 \quad x(0)=0,\dot{x}(0)=0
\]
\item 
\[
\ddot{x} + 2 \dot{x} +2x = e^{-t} \quad x(0)=1,\dot{x}(0)=0
\]
\end{enumerate}
