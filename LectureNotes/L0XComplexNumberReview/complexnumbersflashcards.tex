\cardfrontfoot{Complex Algebra}

\begin{flashcard}[Rectangular to Polar]{$z=a+jb$ \\ Find $z$ in polar form}
\[
z=(\sqrt{a^2+b^2})e^{j\theta}
\]
where
\[
\theta=
\begin{cases}
\tan^{-1}(b/a), & \mbox{if }a\geq 0 \\
\pm\pi + \tan^{-1}(b/a), &  \mbox{if } a < 0
\end{cases}
\]
\begin{itemize}
\item Use four quadrant inverse tangent
\end{itemize}

\end{flashcard}

\begin{flashcard}[Polar to Rectangular]{$z=r\angle\theta=re^{j\theta}$ \\ Find $z$ in rectangular form}
\[
z=r\cos(\theta)+jr\sin(\theta)
\]
\begin{itemize}
\item Use Euler's Formula
\end{itemize}

\end{flashcard}

\begin{flashcard}[Definition]{Euler's Formula}
\[
e^{j\theta}=cos(\theta)+jsin(\theta)
\]

\end{flashcard}

\begin{flashcard}[Exponents]{$e^{a+jb}e^{c+jd}$}
\[
e^{(a+c)+j(b+d)}
\]
\begin{center}
\begin{itemize}
\item
Exponents add
\end{itemize}
\end{center}
\end{flashcard}

\begin{flashcard}[Complex Addition]{$(a+jb)+(c+jd)$}
\[
(a+c)+j(b+d)
\]
\begin{itemize}
\item
Real parts add and imaginary parts add.
\item
Convert polar form to rectangular form before adding.
\end{itemize}
\end{flashcard}

\begin{flashcard}[Complex Multiplication]{$z_1=ae^{j\theta}\,\,\,z_2=be^{j\phi}$\\Find $z_1z_2$}
\[
abe^{j(\theta+\phi)}
\]
\begin{itemize}
\item
Magnitudes multiply, angles add.
\end{itemize}
\end{flashcard}

\begin{flashcard}[Complex Subtraction]{$(a+jb)-(c+jd)$}
\[
(a-c)+j(b-d)
\]
\begin{itemize}
\item
Real parts subtract and imaginary parts subtract.
\item
Convert polar form to rectangular form before subtracting.
\end{itemize}
\end{flashcard}

\begin{flashcard}[Complex Division]{$z_1=ae^{j\theta}\,\,\,z_2=be^{j\phi}$\\Find $\frac{z_1}{z_2}$}
\[
\frac{a}{b}e^{j(\theta-\phi)}
\]
\begin{itemize}
\item
Magnitudes divide and angles subtract.
\end{itemize}
\end{flashcard}

\begin{flashcard}[Roots]{$z^{n}=re^{j\theta}$\\ Solve for $z$}
\[
z=r^{1/n}e^{j\left(\frac{\theta}{n}+\frac{2\pi\ell}{n}\right)} \,\,\, \ell=0,1,\dots,n
\]
\begin{itemize}
\item
n different solutions
\end{itemize}
\end{flashcard}

\begin{flashcard}[Conjugate]{$z=a+jb$ \\ Find $z^{*}$}
\[
z^{*}=a-jb
\]
\begin{itemize}
\item
Imaginary part changes sign
\end{itemize}
\end{flashcard}
